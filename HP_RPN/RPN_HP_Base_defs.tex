% Found on the internet :-)
% Draw a box around a text or letter (for function buttons [f] and [g])
\makeatletter
\newcommand{\sqbox}[1]{%
    \@begin@tempboxa\hbox{#1}%
    \@tempdima=\dimexpr\width-\totalheight\relax
    \ifdim\@tempdima<\z@
        \fbox{\hbox{\hspace{-.1\@tempdima}\usebox\@tempboxa\hspace{-.1\@tempdima}}}%
    \else
        \ht\@tempboxa=\dimexpr\ht\@tempboxa+.1\@tempdima\relax
        \dp\@tempboxa=\dimexpr\dp\@tempboxa+.1\@tempdima\relax
        \fbox{\usebox\@tempboxa}%
    \fi
    \@end@tempboxa
}
\makeatother
